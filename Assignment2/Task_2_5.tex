\begin{align*}
^{i-1}A_i &= 
\begin{pmatrix}
cos(\theta_i) & -sin(\alpha_i) & sin(\theta_i)sin(\alpha_i) & a_icos(\theta_i) \\
sin(\theta_i) & cos(\theta_i)cos(\alpha_i) & -cos(\theta_i)sin(\alpha_i) & a_isin(\theta_i) \\
0 & sin(\alpha_i) & cos(\alpha_i) & d_i \\
0 & 0 & 0 & 1 \\
\end{pmatrix}
\end{align*}
